/**
 * Gets the entry in regionInfo corresponding to the the given encoded region name. If the region
 * has not been seen yet, a new entry is added and returned.
 */
private synchronized HbckRegionInfo getOrCreateInfo(String name) {
    HbckRegionInfo hbi = regionInfoMap.get(name);
    if (hbi == null) {
        hbi = new HbckRegionInfo(null);
        regionInfoMap.put(name, hbi);
    }
    return hbi;
}
